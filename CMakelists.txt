cmake_minimum_required(VERSION 3.10)
project(KeyValueEngine C)

set(CMAKE_C_STANDARD 99)

add_library(kve SHARED kve.c)

add_executable(kve_bin main.c)
set_target_properties(kve_bin PROPERTIES OUTPUT_NAME kve)

target_link_libraries(kve_bin kve)

if(UNIX)
    set_target_properties(kve_bin PROPERTIES LINK_FLAGS "-Wl,-rpath,.")
else()
    set_target_properties(kve_bin PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}")
    set_target_properties(kve PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}")
endif()
